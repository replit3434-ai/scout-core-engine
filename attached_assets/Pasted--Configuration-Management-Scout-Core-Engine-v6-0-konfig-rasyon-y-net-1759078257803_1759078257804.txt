    def __init__(self, config_path: str = "config.json"):
        """
        Konfig√ºrasyonu dosyadan y√ºkle
        
        Args:
            config_path: Konfig√ºrasyon dosyasƒ± yolu
        """
        self.config_path = config_path
        self._load_config()
    
    def _load_config(self):
        """Konfig√ºrasyon dosyasƒ±nƒ± y√ºkle"""
        try:
            config_file = Path(self.config_path)
            
            if config_file.exists():
                with open(config_file, 'r', encoding='utf-8') as f:
                    config_data = json.load(f)
                
                # Konfig√ºrasyon verilerini sƒ±nƒ±f attribute'larƒ±na ata
                for key, value in config_data.items():
                    if hasattr(self, key):
                        setattr(self, key, value)
                
                print(f"‚úÖ Konfig√ºrasyon y√ºklendi: {self.config_path}")
            else:
                # Default konfig√ºrasyon dosyasƒ± olu≈ütur
                self._create_default_config()
                print(f"üìù Default konfig√ºrasyon olu≈üturuldu: {self.config_path}")
                
        except Exception as e:
            print(f"‚ùå Konfig√ºrasyon y√ºkleme hatasƒ±: {e}")
            print("üîÑ Default deƒüerler kullanƒ±lƒ±yor")
    
    def _create_default_config(self):
        """Default konfig√ºrasyon dosyasƒ± olu≈ütur"""

                "log_backup_count": self.log_backup_count,

                "database": self.database,
                "notifications": self.notifications
            }
            
            with open(self.config_path, 'w', encoding='utf-8') as f:
                json.dump(default_config, f, indent=4, ensure_ascii=False)
                
        except Exception as e:
            print(f"‚ùå Default konfig√ºrasyon olu≈üturma hatasƒ±: {e}")
    
    def save_config(self):
        """Mevcut konfig√ºrasyonu dosyaya kaydet"""
        try:
            config_data = {

                "log_backup_count": self.log_backup_count,
                "performance": self.performance,
                "database": self.database,

            }
            
            with open(self.config_path, 'w', encoding='utf-8') as f:
                json.dump(config_data, f, indent=4, ensure_ascii=False)

            print(f"‚ùå Konfig√ºrasyon kaydetme hatasƒ±: {e}")
    
    def update_setting(self, key: str, value: Any):
        """
        Tek bir ayarƒ± g√ºncelle
        
        Args:
            key: Ayar anahtarƒ±
            value: Yeni deƒüer
        """
        try:
            if hasattr(self, key):
                setattr(self, key, value)
                print(f"‚úÖ Ayar g√ºncellendi: {key} = {value}")
            else:
                print(f"‚ùå Bilinmeyen ayar: {key}")
                
        except Exception as e:
            print(f"‚ùå Ayar g√ºncelleme hatasƒ±: {e}")
    
    def get_market_config(self, market_name: str) -> Dict:
        """
        Belirli bir market i√ßin konfig√ºrasyon al
        
        Args:
            market_name: Market adƒ±
            
        Returns:
            Market konfig√ºrasyonu
        """
        return self.market_settings.get(market_name, {})
    
    def is_market_enabled(self, market_name: str) -> bool:
        """
        Market'in aktif olup olmadƒ±ƒüƒ±nƒ± kontrol et
        
        Args:
            market_name: Market adƒ±
            
        Returns:
            Market aktif mi?
        """
        market_config = self.get_market_config(market_name)
        return market_config.get('enabled', False)
    
    def is_league_supported(self, league_id: int) -> bool:
        """
        Lig'in desteklenip desteklenmediƒüini kontrol et
        
        Args:
            league_id: Lig ID'si
            
        Returns:
            Lig destekleniyor mu?
        """
        return league_id in self.supported_leagues
    
    def get_api_credentials(self) -> Dict[str, str]:
        """
        API kimlik bilgilerini al
        
        Returns:
            API credentials
        """
        return {
            'sportmonks_api_key': self.sportmonks_api_key,
            'footystats_api_key': self.footystats_api_key
        }
    
    def validate_config(self) -> List[str]:
        """
        Konfig√ºrasyonu doƒürula
        
        Returns:
            Hata listesi (bo≈üsa konfig√ºrasyon ge√ßerli)
        """
        errors = []
        
        # API key kontrol√º
        if not self.sportmonks_api_key or self.sportmonks_api_key == "YOUR_SPORTMONKS_API_KEY":
            errors.append("SportMonks API key eksik veya ge√ßersiz")
        
        if not self.footystats_api_key or self.footystats_api_key == "YOUR_FOOTYSTATS_API_KEY":
            errors.append("FootyStats API key eksik veya ge√ßersiz")
        
        # Interval kontrol√º
        if self.update_interval < 10:
            errors.append("Update interval √ßok d√º≈ü√ºk (minimum 10 saniye)")
        
        if self.signal_ttl < 5:
            errors.append("Signal TTL √ßok d√º≈ü√ºk (minimum 5 dakika)")
        
        # League kontrol√º
        if not self.supported_leagues:
            errors.append("Desteklenen lig listesi bo≈ü")
        
        # Market kontrol√º
        enabled_markets = [
            market for market, config in self.market_settings.items()
            if config.get('enabled', False)
        ]
        
        if not enabled_markets:
            errors.append("Hi√ßbir market aktif deƒüil")
        
        return errors
    
    def get_summary(self) -> Dict[str, Any]:
        """
        Konfig√ºrasyon √∂zeti al
        
        Returns:
            Konfig√ºrasyon √∂zeti
        """
        enabled_markets = [
            market for market, config in self.market_settings.items()
            if config.get('enabled', False)
        ]
        
        return {
            'api_configured': bool(self.sportmonks_api_key and self.footystats_api_key),
            'update_interval': self.update_interval,
            'signal_ttl': self.signal_ttl,
            'supported_leagues_count': len(self.supported_leagues),
            'enabled_markets': enabled_markets,
            'enabled_markets_count': len(enabled_markets),
            'log_level': self.log_level,
            'rl_enabled': True,
            'notifications_enabled': self.notifications.get('enabled', False)
        }
